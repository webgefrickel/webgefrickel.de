            var mq = document.getElementById('modern-browsers');
            return window.matchMedia && window.matchMedia(mq.media).matches;
        };
        var loadJs = function(src) {
            var ref = document.getElementsByTagName('script')[0];
            var script = document.createElement('script');
            script.src = src;
            script.async = true;
            ref.parentNode.insertBefore(script, ref);
        };
        var loadCss = function(src) {
            var ref = document.getElementsByTagName('head')[0];
            var style = document.createElement('link');
            style.href = src;
            style.rel = 'stylesheet';
            head.append(style);
        };
        if (modern()) {
            loadJs("../../scripts/main.min.js?v=");
        } else {
            loadCSS("../../styles/main.legacy.css?v=");
            loadJs("../../scripts/main.legacy.min.js?v=");
